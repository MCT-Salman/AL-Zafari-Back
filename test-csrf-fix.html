حلول CSRF المقترحة

الحل الأول: تعطيل CSRF للـ API // في app.js app.use((req, res, next) => { if (req.path.startsWith('/notifications') || req.path.startsWith('/auth') || req.headers['content-type']?.includes('application/json')) { return next(); } csrfProtection(req, res, next); });
الحل الثاني: تعديل CSRF middleware const csrfProtection = csrf({ cookie: true, ignoreMethods: ['GET', 'HEAD', 'OPTIONS', 'PUT', 'POST'], value: (req) => { if (req.path.startsWith('/api/') || req.path.startsWith('/notifications')) { return null; } return req.body._csrf; } });
